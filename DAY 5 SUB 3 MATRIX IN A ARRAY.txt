#include <stdio.h>
#define ROWS 3
#define COLS 3
int main() {
    int A[ROWS][COLS], B[ROWS][COLS], C[ROWS][COLS], Result[ROWS][COLS];
    int i, j;
    printf("Enter elements of Matrix A (%dx%d):\n", ROWS, COLS);
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            scanf("%d", &A[i][j]);
        }
    }
    printf("Enter elements of Matrix B (%dx%d):\n", ROWS, COLS);
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            scanf("%d", &B[i][j]);
        }
    }
    printf("Enter elements of Matrix C (%dx%d):\n", ROWS, COLS);
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            scanf("%d", &C[i][j]);
        }
    }
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            Result[i][j] = A[i][j] - B[i][j] - C[i][j];
        }
    }
    printf("Resultant Matrix after subtraction (A - B - C):\n");
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            printf("%d ", Result[i][j]);
        }
        printf("\n");
    }
    return 0;
}
