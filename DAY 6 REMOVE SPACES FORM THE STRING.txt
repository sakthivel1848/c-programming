#include <stdio.h>
int main() {
    char str[200], clean[200];
    int i, j = 0;
    printf("Enter a string: ");
    gets(str); 
    for (i = 0; str[i] != '\0'; i++) {
        if ((str[i] >= 'A' && str[i] <= 'Z') ||
            (str[i] >= 'a' && str[i] <= 'z') ||
            (str[i] >= '0' && str[i] <= '9') ||
            str[i] == ' ') 
        {
            clean[j++] = str[i];
        }
    }
    clean[j] = '\0';
    printf("String after removing shapes/special characters: %s\n", clean);
    return 0;
}
