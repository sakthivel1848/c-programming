#include <stdio.h>
void multiplyMatrices(int first[][10], int second[][10], int result[][10], int row1, int col1, int col2) {
    for (int i = 0; i < row1; i++) {
        for (int j = 0; j < col2; j++) {
            result[i][j] = 0;
            for (int k = 0; k < col1; k++) {
                result[i][j] += first[i][k] * second[k][j];
            }
        }
    }
}
int main() {
    int A[10][10], B[10][10], C[10][10], temp[10][10], result[10][10];
    int r1, c1, r2, c2, r3, c3;
    printf("Enter rows and columns of Matrix A: ");
    scanf("%d %d", &r1, &c1);
    printf("Enter elements of Matrix A:\n");
    for (int i = 0; i < r1; i++)
        for (int j = 0; j < c1; j++)
            scanf("%d", &A[i][j]);
    printf("Enter rows and columns of Matrix B: ");
    scanf("%d %d", &r2, &c2);
    printf("Enter elements of Matrix B:\n");
    for (int i = 0; i < r2; i++)
        for (int j = 0; j < c2; j++)
            scanf("%d", &B[i][j]);
    printf("Enter rows and columns of Matrix C: ");
    scanf("%d %d", &r3, &c3);
    printf("Enter elements of Matrix C:\n");
    for (int i = 0; i < r3; i++)
        for (int j = 0; j < c3; j++)
            scanf("%d", &C[i][j]);
    if (c1 != r2 || c2 != r3) {
        printf("Matrix multiplication not possible with given dimensions.\n");
        return 0;
    }
    multiplyMatrices(A, B, temp, r1, c1, c2);
    multiplyMatrices(temp, C, result, r1, c2, c3);
    printf("Resultant Matrix after multiplying A, B, and C:\n");
    for (int i = 0; i < r1; i++) {
        for (int j = 0; j < c3; j++) {
            printf("%d ", result[i][j]);
        }
        printf("\n");
    }
    return 0;
}
