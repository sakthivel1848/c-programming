#include <stdio.h>
int main() {
    char str[200];
    int i = 0, count = 0;
    printf("Enter a string: ");
    gets(str); 
    while (str[i] != '\0') {
        if ((str[i] != ' ' && str[i] != '\t' && str[i] != '\n') &&
            (i == 0 || str[i - 1] == ' ' || str[i - 1] == '\t' || str[i - 1] == '\n')) {
            count++;
        }
        i++;
    }
    printf("Total words = %d\n", count);
    return 0;
}
